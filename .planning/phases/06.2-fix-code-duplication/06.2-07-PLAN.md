## Implementation

```typescript
// app/lib/csv-spirits.ts

export const SPIRIT_COUNT = 6;

export function extractSpiritsFromRow(row: CSVRow): SpiritEntry[] {
  return Array.from({ length: SPIRIT_COUNT }, (_, i) => ({
    name: row[`spirit${i + 1}`] || "",
    variant: row[`spirit${i + 1}_variant`] || "",
    player: row[`spirit${i + 1}_player`] || "",
  })).filter((s) => s.name);
}

export function spreadSpiritsToColumns(spirits: SpiritEntry[]): Record<string, string> {
  const result: Record<string, string> = {};
  for (let i = 0; i < SPIRIT_COUNT; i++) {
    const s = spirits[i];
    result[`spirit${i + 1}`] = s?.name ?? "";
    result[`spirit${i + 1}_variant`] = s?.variant ?? "";
    result[`spirit${i + 1}_player`] = s?.player ?? "";
  }
  return result;
}
```
